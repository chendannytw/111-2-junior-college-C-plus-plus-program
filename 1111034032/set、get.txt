#include <iostream>
#include <string>
using namespace std;

class DogSystem {
public:
    void setName(string name) {
        dogName = name;
    }

    string getName() const {
        return dogName;
    }

    void setAge(int age) {
        dogAge = age;
    }

    int getAge() const {
        return dogAge;
    }

    void printMessage() const {
        cout << "狗的名字叫做" << getName() << "，年齡是" << getAge() << "歲。" << endl;
    }

private:
    string dogName;
    int dogAge;
};

int main() {
    string nameOfDog;
    int ageOfDog;
    DogSystem myDog;

    cout << "狗系統：" << endl;
    cout << "請輸入狗的名字：" << endl;
    getline(cin, nameOfDog);
    myDog.setName(nameOfDog);

    cout << "請輸入狗的年齡：" << endl;
    cin >> ageOfDog;
    myDog.setAge(ageOfDog);
    if (ageOfDog <= 20){ 
        myDog.printMessage();
    }
    else {
        cout << "糙你擱這騙誰呢" << endl;
    }
    return 0;
}
