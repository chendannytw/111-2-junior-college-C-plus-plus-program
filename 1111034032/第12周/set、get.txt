#include <iostream>
#include <string>
using namespace std;

class DogSystem {
public:
    void setName(string name) {
        dogName = name;
    }

    string getName() {
        return dogName;
    }

    void setAge(int age) {
        dogAge = age;
    }

    int getAge() {
        return dogAge;
    }

    void printMessage() {
        cout << "狗的名字叫做" << getName() << "，年齡是" << getAge() << "歲。" << endl;
    }

private:
    string dogName="Unknown";
    int dogAge = 0;
};

int main() {
    string nameOfDog;
    int ageOfDog;
    DogSystem myDog;

    cout << "狗原本的名字：" << myDog.getName() << endl;
    cout << "請輸入狗的名字：" << endl;
    getline(cin, nameOfDog);
    myDog.setName(nameOfDog);

    cout << "狗原本的年齡：" << myDog.getAge() << endl;
    cout << "請輸入狗的年齡：" << endl;
    cin >> ageOfDog;
    myDog.setAge(ageOfDog);
    if ((ageOfDog <= 20)||(ageOfDog > 0)) {
        myDog.printMessage();
    }
    else {
        cout << "糙你擱這騙誰呢" << endl;
    }
    return 0;
}
